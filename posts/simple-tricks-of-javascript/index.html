کد سیتیAlpine.jsJavaScript

ترفند های ساده JavaScript که زندگی شما رو راحت تر میکنه!

توی این پست میخوام ترفند هایی که خودم خیلی استفاده می کنم و نکاتی که باعث شد برنامه نویس جاوا اسکریپت بهتری بشم رو در اختیار شما قرار بدم.

فهرست مطلب:

  • استفاده از Destructuring assignment
  • استفاده از Spread operator برای کپی کم عمق (Shallow Copy)
  • استفاده از Template literals یا ``
  • اندازه گیری زمان اجرای کد با console.time و تعداد اجرای کد با console.count
  • راه های ساده تبدیل عدد به متن و متن به عدد
  • تبدیل به بولی (boolean) در جاوا اسکریپت
  • به دست آوردن آیتم های منحصر به فرد آرایه
  • پارامتر دوم Array.from همون map هست!
  • استفاده از length برای تغییر طول آرایه
  • حلقه روی آرایه با for of و حلفه روی آبجکت با for in
  • استفاده از Promise.all برای اجرای کد بعد از چندین Promise
  • استفاده از Optional chaining operator
  • تبدیل نوع آیتم های داخل آرایه (مثلا اعداد به رشته ها)

# استفاده از Destructuring assignment

let [a, b] = [10, 20];
console.log(a); // لاگ: 10
console.log(b); // لاگ: 20

let rest;
[a, b, ...rest] = [10, 20, 30, 40, 50];
console.log(rest); // لاگ: [30,40,50]

const obj = { foo: 1, bar: 2, baz: 3 };
const { bar } = obj;
console.log(bar); // لاگ: 2

یه تکنیک جالب هم اینه که مقدار دوتا متغیر رو راحت باهم عوض کنیم:

let a = 1, b = 2;
[a, b] = [b, a];
console.log(a); // لاگ: 2
console.log(b); // لاگ: 1

همچنین میشه موقع تعریف ورودی یا دادن ورودی توابع هم استفاده کرد که واقعا عالی هست:

const device = ({ type, os, price }) => [type, os, price];
console.log(
  device({ type: 'phone', price: '💰💰💰' })
);
// لاگ: ['phone',undefined,'💰💰💰']

# استفاده از Spread operator برای کپی کم عمق (Shallow Copy)

کپی کم عمق آرایه یعنی اینکه خود آرایه به صورت مقدار کپی خواهد شد ولی آیتم های داخل آرایه به صورت رفرنس کپی خواهند شد.

برای درک بهتر کپی رفرنس (pass by reference) و کپی مقدار (pass by value) به تصویر زیر توجه کنید:

let items = [1, 2, 3, 4];
let itemsCopy = [...items];
items.splice(2, 2);
console.log(items) // [1,2]
console.log(itemsCopy) // [1,2,3,4]

فرقش با کپی آرایه به صورت رفرنس:

let items = [1, 2, 3, 4];
let itemsCopy = items;
items.splice(2, 2);
console.log(items) // [1,2]
console.log(itemsCopy) // [1,2]

# استفاده از Template literals یا ``

در جاوا اسکرپیت میتونیم از Template literals برای ساخت رشته (string) استفاده کنیم که هم چند خط بودن رو پشتیبانی میکنه و هم میشه داخلش متغییر قرار داد.

const device = ({ type, os }) => `
  Type: ${type},
  Operating System: ${os}
`;
console.log(device({ type: 'phone', os: 'android' }));
// لاگ: "\n  Type: phone,\n  Operating System: android\n"

# اندازه گیری زمان اجرای کد با console.time و تعداد اجرای کد با console.count

const arr = 'a'.repeat(10000).split('');
console.time('⛏');
for (let i = 0; i < arr.length; i++) {
  arr.find(a => a === i);
}
console.timeEnd('⛏');
// ⛏: 753.705078125 ms

// مثلا تابعی که انتظار داریم 1 بار اجرا بشه
const initState = () => {
  console.count('init called');
}
// یجایی تو پروژه
initState();
// یجای دیگه تو پروژه
initState();

// با این لاگ می فهمیم 2 بار اجرا شد
// init called: 2

# راه های ساده تبدیل عدد به متن و متن به عدد

let num = 45.7;
let str = '45.7';

// تبدیل عدد به متن
console.log(num + ''); // String '45.7'
console.log(`${num}`); // String '45.7'
console.log(String(num)); // String '45.7'
console.log(num.toString()); // String '45.7'

// تبدیل متن به عدد
console.log(+str) // Number 45.7
console.log(Number(str)) // Number 45.7
// عدد صحیح
console.log(~~str) // Number 45
console.log(Math.trunc(str)) // Number 45
console.log(Math.round(str)) // Number 46

# تبدیل به بولی (boolean) در جاوا اسکریپت

استفاده از اوپراتور ! دوبار پشت سر هم

!!486       // true
!![]        // true
!!Infinity  // true
!!''        // false
!!undefined // false
!!NaN       // false

و یا استفاده از Boolean:

Boolean(1)  // true
Boolean({}) // true
Boolean(0)  // false

همچنین خوبه که بدونیم مقادیر Falsy و Truthy کدوما هستند.

نمونه مقادیر Falsy:

// همه اینا فالس هستند
if (false)
if (null)
if (undefined)
if (0)
if (-0)
if (0n)
if (NaN)
if ("")
if (document.all) // 😂 این یکی داستان داره: https://stackoverflow.com/questions/10350142

نمونه مقادیر Truthy:

// همه اینا ترو هستند
if (true)
if ({})
if ([])
if (42)
if ("0")
if ("false")
if (new Date())
if (-42)
if (12n)
if (3.14)
if (-3.14)
if (Infinity)
if (-Infinity)

# به دست آوردن آیتم های منحصر به فرد آرایه

خیلی ساده با استفاده از Set که ذاتا منحصر به فرد هست و خود به خود آیتم های تکراری رو حذف میکنه، میتونیم آیتم های منحصر به فرد رو به دست بیاریم

let arr = ['💩', '💩', '💩', '🌼', '🌼', '🌼', '🌴', '🌴', '🌴'];

let uniqueArr = [...new Set(arr)];

console.log(uniqueArr); // لاگ: ['💩','🌼','🌴']

# پارامتر دوم Array.from همون map هست!

یعنی مثلا بجای:

let array = Array.from(document.querySelectorAll('a'));
let links = array.map(a => a.href)

از این هم میشه استفاده کرد:

let links = Array.from(document.querySelectorAll('a'), a => a.href);

# استفاده از length برای تغییر طول آرایه

مثلا دو آیتم از آخر رو حذف می کنیم:

let arr = [1, 2, 3, 4];
arr.length = 2;
console.log(arr); // [1, 2]

# حلقه روی آرایه با for of و حلفه روی آبجکت با for in

// حلقه راحت روی آرایه
for (let item of myArray) {
}

// حلقه راحت روی آبجکت
for (let key in myObject) {
}

// حلقه روی آبجک وقتی هم کلید و هم مقدار رو میخوایم 
for (let [key, value] of Object.entries(obj)) {
}

# استفاده از Promise.all برای اجرای کد بعد از چندین Promise

const promise1 = Promise.resolve(3);
const promise2 = new Promise(resolve => {
  setTimeout(resolve, 500, 'foo');
});
const promise3 = 45;

Promise.all([promise1, promise2, promise3])
.then(values => {
  console.log(values);
});
// لاگ: Array [3, 'foo', 42]

# استفاده از Optional chaining operator

این یکی رو خیلی لازم داشتیم توی جاوا اسکریپت! تا قبل از این اگر میخواستیم به صورت عمیق به داخل یک آبجکت دسترسی داشته باشیم و موقع اینکار یکی از property ها undefined بود، کلا جاوا اسکریپت خطا می داد و تنها راه استفاده از || بود که یکم زحمت داشت ولی با استفاده از Optional chaining دیگه این مشکل رفع شد:

const device = {
  type: 'phone',
  os: {
    name: 'android',
    version: '12',
  },
};

let cpuModel;

cpuModel = device.hardware.cpu.model; // خطا!

cpuModel = (((device || {}).hardware || {}).cpu || {}).model // قبل از Optional chaining

cpuModel = device?.hardware?.cpu?.model // بعد از Optional chaining

// حتی برای توابع هم میشه استفاده کرد
device?.camera?.startRecordingVideo?.();
// بجای همچین چیزی:
device && device.camera && device.camera.startRecordingVideo && device.camera.startRecordingVideo();

# تبدیل نوع آیتم های داخل آرایه (مثلا اعداد به رشته ها)

let items = [1, 0, 'true', null, {}, [], true, undefined, NaN, '0', false];

items.map(Boolean); // [true, false, true, false, true, true, true, false, false, true, false]

items.map(Number); // [1, 0, NaN, 0, NaN, 0, 1, NaN, NaN, 0, 0]

items.map(String); // ["1", "0", "true", "null", "[object Object]", "", "true", "undefined", "NaN", "0", "false"]
